void pt(vector<pp> &l)
{

    cout<<l[i].first<<' '<<l[i].second<<endl;
  cout<<endl;
}
void cm(vector<pp> &c)
{
  if(c.size()<2)
    return ;
  int nn=0;
  for(int j=1;j<c.size();++j)
    if(c[nn].second+1==c[j].first)
      c[nn].second=c[j].second;
    else
      {
	++nn;
	if(nn<j)
	  c[nn]=c[j];
      }
  c.resize(nn+1);
}
